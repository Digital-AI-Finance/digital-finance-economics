% Payment System Basics
\begin{frame}[t]{Economics of Payment Systems}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Economic Functions}

Payment systems enable:
\begin{itemize}
\item Value transfer between parties
\item Settlement of obligations
\item Support for economic transactions
\end{itemize}

\vspace{0.3em}
\textbf{Key Economic Properties}
\begin{itemize}
\item Network goods (value $\uparrow$ with users)
\item Infrastructure characteristics
\item Significant fixed costs
\end{itemize}

\column{0.48\textwidth}
\textbf{System Types}

\vspace{0.3em}
\textbf{Large Value (Wholesale)}
\begin{itemize}
\item Fedwire, TARGET2
\item RTGS: Real-time gross settlement
\item Low volume, high value
\end{itemize}

\vspace{0.3em}
\textbf{Retail}
\begin{itemize}
\item Cards, ACH, instant payments
\item High volume, lower value
\item Consumer-facing
\end{itemize}
\end{columns}

\bottomnote{Payment systems exhibit natural monopoly characteristics due to network effects}
\end{frame}

% Network Effects
\begin{frame}[t]{Network Effects in Payments}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Direct Network Effects}

Value increases with users:
$$V(n) = n \cdot v(n)$$

where $v(n)$ is per-user value.

\vspace{0.3em}
\textbf{Metcalfe's Law (simplified)}
$$V \propto n^2$$
\begin{itemize}
\item Each user can transact with $n-1$ others
\item Creates positive feedback loops
\end{itemize}

\column{0.48\textwidth}
\textbf{Implications for Payments}

\vspace{0.3em}
\textbf{Adoption Dynamics}
\begin{itemize}
\item Critical mass threshold
\item Tipping points
\item Winner-take-most markets
\end{itemize}

\vspace{0.3em}
\textbf{Entry Barriers}
\begin{itemize}
\item Incumbents have user base advantage
\item New entrants need to ``buy'' network
\item Interoperability can reduce barriers
\end{itemize}
\end{columns}

\bottomnote{Network effects explain why few payment networks dominate each market segment}
\end{frame}

% Two-Sided Markets
\begin{frame}[t]{Two-Sided Market Theory}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{The Platform Model}

Card networks connect:
\begin{itemize}
\item Side 1: Cardholders (consumers)
\item Side 2: Merchants (acceptors)
\item Platform: Visa, Mastercard
\end{itemize}

\vspace{0.3em}
\textbf{Cross-Side Network Effects}
\begin{itemize}
\item More merchants $\rightarrow$ more cardholders
\item More cardholders $\rightarrow$ more merchants
\item Chicken-and-egg problem
\end{itemize}

\column{0.48\textwidth}
\textbf{Rochet-Tirole Model}

Optimal pricing:
$$p_B + p_S = c + m$$
$$\frac{p_B - c_B}{p_S - c_S} = \frac{\eta_S}{\eta_B}$$

where $\eta$ = price elasticity

\vspace{0.3em}
\textbf{Key Insight}

Price structure matters, not just level:
\begin{itemize}
\item Subsidize price-sensitive side
\item Charge price-insensitive side
\item ``Get both sides on board''
\end{itemize}
\end{columns}

\bottomnote{Rochet \& Tirole (2006): Two-sided markets require analysis beyond standard economics}
\end{frame}

% Interchange Fees
\begin{frame}[t]{Interchange Fee Economics}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{The Fee Flow}

\begin{enumerate}
\item Consumer pays \$100
\item Merchant receives \$97-98
\item Interchange: 1.5-2\% to issuer
\item Network fee: 0.1-0.2\%
\item Acquirer margin: 0.2-0.5\%
\end{enumerate}

\vspace{0.3em}
\textbf{Economic Rationale}
\begin{itemize}
\item Issuer bears fraud risk
\item Subsidizes cardholder rewards
\item Balances two-sided market
\end{itemize}

\column{0.48\textwidth}
\textbf{Regulatory Debate}

\vspace{0.3em}
\textbf{Against High Interchange}
\begin{itemize}
\item Merchants pass costs to prices
\item Regressive: cash users subsidize card users
\item Anti-competitive coordination
\end{itemize}

\vspace{0.3em}
\textbf{For Market Rates}
\begin{itemize}
\item Funds card benefits
\item Network competition exists
\item Caps reduce innovation
\end{itemize}
\end{columns}

\bottomnote{EU capped interchange at 0.2-0.3\%; US Durbin Amendment capped debit at 0.05\%+21c}
\end{frame}

% Cross-Border Payments
\begin{frame}[t]{Cross-Border Payment Challenges}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{The Problem}

Cross-border payments are:
\begin{itemize}
\item Expensive: 6\%+ average cost
\item Slow: 2-5 days settlement
\item Opaque: uncertain fees
\item Fragmented: many intermediaries
\end{itemize}

\vspace{0.3em}
\textbf{Root Causes}
\begin{itemize}
\item Lack of common infrastructure
\item Regulatory fragmentation
\item Legacy technology
\item Correspondent banking model
\end{itemize}

\column{0.48\textwidth}
\textbf{Economic Inefficiencies}

\vspace{0.3em}
\textbf{FX Costs}
\begin{itemize}
\item Wide bid-ask spreads
\item Hidden markups in rates
\item Multiple conversions
\end{itemize}

\vspace{0.3em}
\textbf{Compliance Costs}
\begin{itemize}
\item AML/KYC at each hop
\item Sanctions screening
\item Data format inconsistencies
\end{itemize}
\end{columns}

\bottomnote{G20 target: reduce average cost to 3\% by 2027; currently at 6\%+}
\end{frame}

% Settlement Systems
\begin{frame}[t]{Settlement System Design}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{RTGS (Real-Time Gross)}

Characteristics:
\begin{itemize}
\item Immediate, final settlement
\item Each transaction settled individually
\item High liquidity requirement
\end{itemize}

\vspace{0.3em}
Examples: Fedwire, TARGET2

\vspace{0.3em}
Trade-offs:
\begin{itemize}
\item[\textcolor{mlgreen}{+}] Eliminates settlement risk
\item[\textcolor{mlred}{-}] High liquidity cost
\end{itemize}

\column{0.48\textwidth}
\textbf{DNS (Deferred Net)}

Characteristics:
\begin{itemize}
\item Batch settlement at intervals
\item Payments netted against each other
\item Lower liquidity needs
\end{itemize}

\vspace{0.3em}
Examples: ACH, many retail systems

\vspace{0.3em}
Trade-offs:
\begin{itemize}
\item[\textcolor{mlgreen}{+}] Liquidity efficient
\item[\textcolor{mlred}{-}] Settlement risk until batch
\end{itemize}
\end{columns}

\bottomnote{Hybrid systems combine benefits: queue management, partial netting with RTGS}
\end{frame}

% Instant Payments
\begin{frame}[t]{Instant Payments: Economic Impact}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Global Rollout}

Major systems:
\begin{itemize}
\item UK: Faster Payments (2008)
\item India: UPI (2016)
\item EU: SEPA Instant (2017)
\item US: FedNow (2023)
\end{itemize}

\vspace{0.3em}
\textbf{Design Features}
\begin{itemize}
\item 24/7/365 availability
\item Settlement in seconds
\item Irrevocable payments
\end{itemize}

\column{0.48\textwidth}
\textbf{Economic Benefits}

For consumers:
\begin{itemize}
\item Improved cash flow management
\item Emergency transfers
\item P2P payments
\end{itemize}

For businesses:
\begin{itemize}
\item Working capital optimization
\item Reduced float costs
\item Real-time reconciliation
\end{itemize}

For economy:
\begin{itemize}
\item Velocity of money increase
\item Reduced payment friction
\end{itemize}
\end{columns}

\bottomnote{India's UPI: 10+ billion transactions/month; transformed payment landscape}
\end{frame}

% Financial Inclusion
\begin{frame}[t]{Financial Inclusion Economics}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{The Unbanked}

1.4 billion adults lack accounts:
\begin{itemize}
\item Documentation barriers
\item Physical access (branches)
\item Minimum balance requirements
\item Trust and literacy issues
\end{itemize}

\vspace{0.3em}
\textbf{Economic Costs}
\begin{itemize}
\item Check cashing fees (2-5\%)
\item No savings accumulation
\item Excluded from credit
\end{itemize}

\column{0.48\textwidth}
\textbf{Mobile Money Success}

M-Pesa (Kenya) model:
\begin{itemize}
\item Agent network (not branches)
\item Phone-based (no smartphone needed)
\item Low-value, low-cost transactions
\end{itemize}

\vspace{0.3em}
\textbf{Impact Evidence}
\begin{itemize}
\item Suri \& Jack (2016): 2\% poverty reduction
\item Women especially benefited
\item Improved risk sharing
\end{itemize}
\end{columns}

\bottomnote{Mobile money shows technology can reduce barriers; requires complementary ecosystem}
\end{frame}
